removeLast() {
        if (!this.head) {
          return;
        }

        if (!this.head.next) {
          this.head = null;
          return;
        }

        let previous = this.head;
        let node = this.head.next;
        while (node.next) {
          previous = node;
          node = node.next;
        }
        previous.next = null;
      }
